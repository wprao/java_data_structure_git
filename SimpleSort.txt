SimpleSort:

1)BubbleSort:

	for(i=nElems-1;i>0;i--)
	{
		for(j=0;j<=i;j++)
		{
			if(array[j]>array[j+1])
				swap(j,j+1);
		}
	}

2)SelectSort:
	for(i=nElems-1;i>0;i--)
	{
		max = i;
		for(j=0;j<=i;j++)
		{
			if(array[j]>array[max])
			{
				max = j;
			}
		}
		swap(max,j;)
	}

3)InsertSort

	for(i=1;i<nElems;i++)
	{
		temp = array[i];
		j = i;
		while(j>0&&array[j-1]>temp)
		{
			array[j] = array[j-1];
			j--;
		}
		array[j] = temp;
	}
